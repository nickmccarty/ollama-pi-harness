# requirements-vllm.txt — vLLM serving dependencies
#
# IMPORTANT: install in a SEPARATE virtual environment (or use Docker).
# Do NOT pip install into the main harness env — vLLM pins its own torch/triton/
# flashinfer versions that conflict with the harness's torch==2.7.1+cu118.
#
# =============================================================================
# RECOMMENDED: Docker (avoids all build-time conflicts)
# =============================================================================
#
# Pull the pinned release image:
#   docker pull vllm/vllm-openai:v0.19.0
#
# Serve pi-qwen3.6 (Qwen3.6-35B-A3B AWQ, ~17GB VRAM):
#   docker run --gpus all --ipc=host -p 8000:8000 \
#     -v ~/.cache/huggingface:/root/.cache/huggingface \
#     vllm/vllm-openai:v0.19.0 \
#     --model QuantTrio/Qwen3.6-35B-A3B-AWQ \
#     --quantization awq_marlin \
#     --dtype float16 \
#     --max-model-len 32768 \
#     --reasoning-parser qwen3 \
#     --enable-auto-tool-choice \
#     --tool-call-parser qwen3_coder \
#     --enable-prefix-caching \
#     --gpu-memory-utilization 0.90
#
# Serve evaluator alongside (second container, second port):
#   docker run --gpus all --ipc=host -p 8001:8000 \
#     -v ~/.cache/huggingface:/root/.cache/huggingface \
#     vllm/vllm-openai:v0.19.0 \
#     --model Qwen/Qwen2.5-7B-Instruct \
#     --dtype half \
#     --max-model-len 16384 \
#     --enable-prefix-caching \
#     --gpu-memory-utilization 0.24
#
# LoRA hot-swap (integrator agent):
#   Add --enable-lora --max-lora-rank 64 to the startup flags above.
#   Load at runtime: POST /v1/load_lora_adapter {"lora_name": "v2", "lora_path": "..."}
#
# =============================================================================
# Pip install (CUDA 11.8 build — matches harness torch+cu118)
# =============================================================================
#
# In a fresh venv:
#   python -m venv .venv-vllm
#   source .venv-vllm/bin/activate   (or .venv-vllm\Scripts\activate on Windows)
#   pip install -r requirements-vllm.txt
#
# Version policy: pin to a specific release; do NOT use `vllm>=x.y` in production.
# Run `pip-compile --generate-hashes requirements-vllm.in` after any version bump
# to get SHA256-pinned hashes for the full transitive closure.
#
# =============================================================================

# Core serving stack
# vLLM 0.19+ recommends uv for installation (avoids pip wheel build issues on CUDA).
# DO NOT pip install vllm directly — use the uv command below.
#
# Install (WSL2, CUDA 12.x):
#   pip install uv
#   uv pip install vllm --torch-backend=auto   # pulls vllm==0.19.x + matching torch
#
# Then pin transformers (vLLM 0.19 ships with 5.x; 5.5.4 tested with Qwen3.6):
#   pip install "transformers==5.5.4"
#
# flashinfer and triton are now bundled with vLLM — do NOT pin separately.
# ray is only needed for multi-GPU tensor-parallel serving; omit for single-GPU.

# Listed for reference / pip-compile tooling — actual install uses uv above
vllm==0.19.0
transformers==5.5.4

# OpenAI client — used by inference.py to talk to vLLM's /v1 endpoint
# Already in main requirements.txt; listed here for the isolated venv case
openai==2.31.0

# =============================================================================
# Environment variables for the harness when vLLM is active
# =============================================================================
# Add these to .env (they override the defaults in inference.py):
#
#   INFERENCE_BACKEND=vllm
#   VLLM_BASE_URL=http://localhost:8000/v1
#
# To route the evaluator to a second vLLM instance on port 8001, set a custom
# model map entry that the wiggum shim can read:
#   VLLM_MODEL_MAP={"Qwen3-Coder:30b": "Qwen/Qwen2.5-Coder-32B-Instruct"}
#   VLLM_EVALUATOR_BASE_URL=http://localhost:8001/v1  (future — inference.py v2)
#
# =============================================================================
# Known version constraints
# =============================================================================
# vLLM 0.8+ requires CUDA 12.x and does not support native Windows pip installs
# (platform check + MAX_PATH failures during wheel build). Use WSL2 or Docker.
# Tested: vllm==0.19.0 + torch==2.6.0+cu124 in WSL2 Ubuntu 24.04 on RTX 5000 Ada.
# Qwen3.6-35B-A3B-AWQ (awq_marlin) fits in ~17-18GB VRAM at gpu-memory-utilization 0.90.
# Track: https://github.com/vllm-project/vllm/releases
#
# Security incident context (2024 supply-chain): unconstrained `pip install vllm`
# pulled a malicious package via a transitive dep. Mitigation:
#   1. Pin every direct dep above (no bare `vllm` without ==version)
#   2. Use --require-hashes after running pip-compile --generate-hashes
#   3. Prefer the Docker image: it ships a tested+signed dep closure
#   4. Audit diff before any version bump: `pip install pip-audit && pip-audit`
